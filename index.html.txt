<!-- Selector de PDFs adicionales -->
<div id="pdfSection" style="margin-top:30px; display:none;">
    <label for="pdfSelect">Selecciona PDF adicional:</label>
    <select id="pdfSelect">
        <option value="">--Seleccione PDF--</option>
    </select>
    <br>
    <button onclick="openPDF()">Abrir PDF</button>
</div>

<script>
    // PDFs según idioma
    const pdfData = {
        es: {
            Barcos: "pdfs/es/barcos.pdf",
            Actividades en el agua: "pdfs/es/actividades en el agua.pdf",
            Fondos marinos: "pdfs/es/fondos marinos.pdf"
        },
        en: {
            water: "pdfs/en/water.pdf",
            earth: "pdfs/en/earth.pdf",
            air: "pdfs/en/air.pdf"
        },
        de: {
            wasser: "pdfs/de/wasser.pdf",
            erde: "pdfs/de/erde.pdf",
            luft: "pdfs/de/luft.pdf"
        }
    };

    let selectedLang = "";

    function selectLanguage(lang) {
        selectedLang = lang;
        // Mostrar el selector de PDFs adicionales
        const pdfSection = document.getElementById("pdfSection");
        pdfSection.style.display = "block";

        // Rellenar el selector con PDFs según idioma
        const pdfSelect = document.getElementById("pdfSelect");
        pdfSelect.innerHTML = '<option value="">--Seleccione PDF--</option>';
        for (const key in pdfData[lang]) {
            const option = document.createElement("option");
            option.value = pdfData[lang][key];
            option.textContent = key.charAt(0).toUpperCase() + key.slice(1);
            pdfSelect.appendChild(option);
        }

        // Abrir el PDF principal de idioma
        const mainPdf = {
            es: "pdfs/es/documento_es.pdf",
            en: "pdfs/en/documento_en.pdf",
            de: "pdfs/de/documento_de.pdf"
        };
        window.open(mainPdf[lang], "_blank");
    }

    function openPDF() {
        const pdfSelect = document.getElementById("pdfSelect");
        const pdfUrl = pdfSelect.value;
        if (pdfUrl) {
            window.open(pdfUrl, "_blank");
        } else {
            alert("Por favor selecciona un PDF.");
        }
    }
</script>
